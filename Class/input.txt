// Comentario de inicio del programa

class Program {
  // Declaración de variables de instancia
  int x; // variable entera
  boolean y; // variable booleana
  int[] z; // array de enteros

  // Método principal del programa
  void main() {
    // Asignación de valores a variables
    x = 5; // asignar 5 a x
    y = true; // asignar true a y
    z = new int[10]; // crear un array de 10 enteros

    // Uso de condicionales (if-else)
    if (x > 5) { // si x es mayor que 5
      println("x es mayor que 5"); // imprimir mensaje
    } else { // si no
      println("x es menor o igual que 5"); // imprimir mensaje
    }

    // Uso de bucles (for)
    for (int i = 0; i < 10; i++) { // iterar de 0 a 9
      z[i] = i * 2; // asignar valor a cada elemento del array
    }

    // Uso de bucles (while)
    int j = 0; // inicializar variable
    while (j < 10) { // mientras j sea menor que 10
      println(z[j]); // imprimir elemento del array
      j++; // incrementar j
    }

    // Llamada a método
    metodo1(5, true); // llamar a metodo1 con parámetros

    // Retorno de valor
    return x + y; // retornar suma de x e y
  }

  // Método con parámetros
  void metodo1(int a, boolean b) {
    // Imprimir mensaje con parámetros
    println("metodo1 llamado con a = " + a + " y b = " + b);
  }

  // Método sin parámetros
  void metodo2() {
    // Imprimir mensaje
    println("metodo2 llamado");
  }

  // Método con retorno de valor
  int metodo3(int a, int b) {
    // Retornar suma de parámetros
    return a + b;
  }

  // Método con retorno de valor booleano
  boolean metodo4(boolean a, boolean b) {
    // Retornar resultado de operación lógica
    return a && b;
  }
}
